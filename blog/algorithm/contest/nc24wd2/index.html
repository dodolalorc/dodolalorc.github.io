    void solve() {
    int n, m, k;cin >> n >> m >> k;
    map<pair<int, int>, bool>cats;
    int ans = 0;
    while (k--) {
        int x, y;cin >> x >> y;
        cats[{ x, y }] = true;
        pair<int, int>pu = { x - 1,y }, pd = { x + 1,y }, pl = { x,y - 1 }, pr = { x,y + 1 };
        int cnt = 4;
        if (cats.count(pu) != 0)cnt -= 1;
        if (cats.count(pd) != 0)cnt -= 1;
        if (cats.count(pl) != 0)cnt -= 1;
        if (cats.count(pr) != 0)cnt -= 1;
        ans += cnt;
    }
    cout << ans << endl;
}

    int n;cin >> n;
    vector<ll>a(n);

    for (int i = 0;i < n;i++) {
